/ UNTUK MEMBUAT TTYD BOOT DAN JALAN KETIKA REBOOT.
mkdir -p $HOME/.termux/boot
cd $HOME/.termux/boot
cat > ./ttyd-service.sh <<EOF
#!/data/data/com.termux/files/usr/bin/sh
termux-wake-lock
ttyd -p 3001 -W -t enableTrzsz=true bash
EOF
